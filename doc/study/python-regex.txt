def convert_and_print():
    pat = re.compile(r'array\([^)]*\)\)')
    count = 0

    for match in pat.finditer(province_str):
        onepro = match.group()
        pat1 = re.compile(r'=>.*\'(.*)\',.*=>.*\((.*)\)\)', re.DOTALL)
        match1 = pat1.search(onepro)
        count += 1
        pat2 = re.compile(r'[^\', \n]+')
        citylst = pat2.findall(match1.group(2))
        citystr = "["

        for val in citylst:
            citystr += "\"{}\",".format(val)

        citystr = citystr[:len(citystr) - 1]
        citystr += "];"
        print("provinceDict[\"{}\"] = {}".format(match1.group(1), citystr))

    print("count = {}".format(count))

if __name__ == "__main__":
    convert_and_print()
